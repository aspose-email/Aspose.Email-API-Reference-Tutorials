# วิธีการตั้งค่ารูปแบบการเกิดซ้ำรายเดือนในงาน Outlook โดยใช้ Aspose.Email .NET

## การแนะนำ

คุณกำลังมองหาวิธีทำให้การกำหนดตารางงานของคุณเป็นแบบอัตโนมัติโดยการตั้งค่ารูปแบบการเกิดซ้ำรายเดือนใน Outlook โดยใช้ Aspose.Email สำหรับ .NET หรือไม่ ไม่ว่าคุณจะกำลังจัดการรายการสิ่งที่ต้องทำส่วนตัวหรือประสานงานไทม์ไลน์ของโครงการที่ซับซ้อน งานที่เกิดซ้ำสามารถเพิ่มประสิทธิภาพได้อย่างมาก ในบทช่วยสอนนี้ เราจะมาสำรวจว่าคุณสามารถใช้พลังของ Aspose.Email สำหรับ .NET เพื่อสร้างตารางงานที่สอดคล้องและเชื่อถือได้ได้อย่างไร

**สิ่งที่คุณจะได้เรียนรู้:**
- วิธีการตั้งค่ารูปแบบการเกิดซ้ำรายเดือนในงาน Outlook
- การคำนวณการเกิดขึ้นระหว่างสองวันที่มีกฎการเกิดซ้ำที่ระบุ
- การนำฟังก์ชัน Aspose.Email ไปใช้งานอย่างมีประสิทธิภาพ

เมื่ออ่านคู่มือนี้จบ คุณจะพร้อมสำหรับการกำหนดตารางงานโดยอัตโนมัติโดยไม่ต้องออกแรง มาเจาะลึกข้อกำหนดเบื้องต้นกันก่อนจะเริ่มกันเลย

## คู่มือการใช้งาน

ตอนนี้เราจะแบ่งการใช้งานออกเป็นคุณลักษณะที่แตกต่างกันเพื่อให้เข้าใจได้ดียิ่งขึ้น

### คุณสมบัติ 1: การตั้งค่ารูปแบบการเกิดซ้ำรายเดือน

#### ภาพรวม
ฟีเจอร์นี้สาธิตการตั้งค่ารูปแบบการเกิดซ้ำรายเดือนสำหรับงาน Outlook โดยอนุญาตให้ทำงานซ้ำในวันเฉพาะแต่ละเดือน

#### การดำเนินการแบบทีละขั้นตอน

##### กำหนดวันที่เริ่มต้นและสิ้นสุด
ขั้นแรก ให้กำหนดวันที่เริ่มต้นงานและวันที่ควรสิ้นสุดงาน ปรับวันที่เหล่านี้ตามเขตเวลาท้องถิ่น:

```csharp
using Aspose.Email.Mapi;
using System;

// กำหนดวันที่เริ่มต้นและสิ้นสุดพร้อมการปรับโซนเวลา
DateTime StartDate = new DateTime(2015, 7, 1);
DateTime endByDate = new DateTime(2015, 12, 31);

TimeZone localZone = TimeZone.CurrentTimeZone;
TimeSpan ts = localZone.GetUtcOffset(DateTime.Now);
StartDate = StartDate.Add(ts);
endByDate = endByDate.Add(ts);
```

##### สร้างงาน Outlook ใหม่
สร้างและกำหนดค่างานของคุณ:

```csharp
// สร้าง MapiTask ใหม่
MapiTask task = new MapiTask("This is test task", "Sample Body", StartDate, StartDate);
task.State = MapiTaskState.NotAssigned;
```

##### ตั้งค่ารูปแบบการเกิดซ้ำรายเดือน
กำหนดค่ารายละเอียดรูปแบบการเกิดซ้ำ:

```csharp
var recurrence = new MapiCalendarMonthlyRecurrencePattern {
    Day = 15,
    Period = 1,
    PatternType = MapiCalendarRecurrencePatternType.Month,
    EndType = MapiCalendarRecurrenceEndType.EndAfterDate,
    OccurrenceCount = GetOccurrenceCount(StartDate, endByDate, "FREQ=MONTHLY;BYMONTHDAY=15;INTERVAL=1"),
    WeekStartDay = DayOfWeek.Monday,
    EndDate = endByDate
};
task.Recurrence = recurrence;
```

##### วิธีช่วยเหลือในการคำนวณการเกิดขึ้น

```csharp
private static uint GetOccurrenceCount(DateTime start, DateTime endBy, string rrule) {
    CalendarRecurrence pattern = new CalendarRecurrence(string.Format("DTSTART:{0}\r\nRRULE:{1}", start.ToString("yyyyMMdd"), rrule));
    DateCollection dates = pattern.GenerateOccurrences(start, endBy);
    return (uint)dates.Count;
}
```

### คุณลักษณะที่ 2: การคำนวณจำนวนการเกิดซ้ำ

#### ภาพรวม
คำนวณจำนวนครั้งที่เกิดขึ้นสำหรับกฎการเกิดซ้ำที่กำหนดระหว่างวันที่สองวันที่ระบุ

#### การดำเนินการแบบทีละขั้นตอน

##### คำนวณการเกิดขึ้น
สร้างและกำหนดค่าตรรกะการคำนวณการเกิดซ้ำของคุณ:

```csharp
using Aspose.Email.Calendar.Recurrences;
using System;

public static uint CalculateOccurrences(DateTime start, DateTime endBy, string rrule) {
    CalendarRecurrence pattern = new CalendarRecurrence(string.Format("DTSTART:{0}\r\nRRULE:{1}", start.ToString("yyyyMMdd"), rrule));
    DateCollection dates = pattern.GenerateOccurrences(start, endBy);
    return (uint)dates.Count;
}
```
